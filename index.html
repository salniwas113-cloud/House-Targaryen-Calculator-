// services/pnwApi.js
import axios from 'axios';

const PNW_API = 'https://api.politicsandwar.com/graphql';

export const fetchNation = async (nationId, apiKey) => {
  const query = `{
    nations(id: [${nationId}]) {
      data {
        id nation_name leader_name
        num_cities score
        cities { infrastructure land }
      }
    }
  }`;
  const { data } = await axios.post(
    `${PNW_API}?api_key=${apiKey}`,
    { query }
  );
  return data.data.nations.data[0];
};
